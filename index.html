create pem files

create the private key pem file

openssl genrsa -out private_key.pem 2048

create self signature certificate signing request pem file

openssl req -new -key private_key.pem -out csr.pem

create certificate pem file by self signature

openssl x509 -req -in csr.pem -signkey private_key.pem -out certificate.pem
